# Standardization — same transformation used at training time
scale_apply <- function(df, means, sds, vars) {
  out <- df
  for (v in vars) {
    out[[v]] <- (out[[v]] - means[[v]]) / ifelse(sds[[v]] == 0, 1, sds[[v]])
  }
  out
}

# Recommendation strength classifier
# Logic:
#   Strong Recommend — prob_best >= 0.75 AND lower bound of 80% CI on
#                      edge over second-best is positive.
#                      The model is both confident this is best AND
#                      confident the margin is non-trivial.
#   Recommend        — prob_best >= 0.55 OR expected edge > 0.02 EPA.
#                      Clear preferred package but some uncertainty remains.
#   Light Recommend  — everything else. Posterior distributions substantially
#                      overlap. Coach should weight non-model factors heavily.

classify_strength <- function(prob_best, edge_lower80, expected_edge) {
  if (prob_best >= 0.75 && edge_lower80 > 0) {
    return("Strong Recommend")
  } else if (prob_best >= 0.55 || expected_edge > 0.02) {
    return("Recommend")
  } else {
    return("Light Recommend")
  }
}

1.1 Convergence Diagnostics

Before any predictive metrics, confirm all three models converged. Rhat values above 1.01 indicate the chains did not fully agree — any such parameters should be inspected before trusting recommendations from the model.

Code 
check_convergence <- function(fit, label) {
  rhats <- brms::rhat(fit)
  n_bad <- sum(rhats > 1.01, na.rm = TRUE)
  n_total <- length(rhats)
  if (n_bad == 0) {
    cat(sprintf("  %-12s All %d parameters: Rhat <= 1.01 ✓\n", label, n_total))
  } else {
    cat(sprintf("  %-12s WARNING: %d / %d parameters with Rhat > 1.01\n",
                label, n_bad, n_total))
    bad <- sort(rhats[rhats > 1.01], decreasing = TRUE)
    print(head(bad, 10))
  }
}

1.3 Stage 1 — Pass Probability Calibration

Calibration measures whether predicted probabilities are accurate, not just well-ranked. A perfectly calibrated model produces points on the diagonal: when it predicts 70% pass probability, roughly 70% of those plays are passes.

Deviations above the diagonal mean the model is underestimating pass probability in that range; deviations below mean it is overestimating.

Code 
# Get posterior mean predicted probabilities on test set
p_pass_hat <- colMeans(
  posterior_epred(fit_pass, newdata = test_df, allow_new_levels = TRUE)
)

# Build calibration bins
calib_df <- tibble(
  pred   = p_pass_hat,
  actual = test_df$is_pass
) %>%
  mutate(bin = cut(pred, breaks = seq(0, 1, 0.05), include.lowest = TRUE)) %>%
  group_by(bin) %>%
  summarise(
    mean_pred   = mean(pred),
    mean_actual = mean(actual),
    n           = n(),
    se          = sqrt(mean_actual * (1 - mean_actual) / n),
    .groups = "drop"
  ) %>%
  filter(n >= 20)   # drop bins with too few plays to be reliable

ggplot(calib_df, aes(x = mean_pred, y = mean_actual)) +

1.5 Decision Quality Evaluation

This is the most important evaluation for a prescriptive model. For plays where the model agreed with the coach’s actual package choice versus plays where it strongly disagreed (prob_best > 0.70 for a different package), compare the actual EPA allowed.

This is observational — we cannot observe the counterfactual — but consistent differences in the expected direction provide the strongest available evidence that the recommendations are meaningful.

# ── prepare_scenario ──────────────────────────────────────────────────────────
# Builds a 3-row data frame (one per package) for a given game situation.
# Team-level features can be supplied from real team data or left at league
# average defaults for neutral/unknown opponent analysis.
#
# The matchup_cell and off_personnel_team_season keys are built using the
# same logic as training so brms can look up partial-pooling estimates.
# New 2025 matchup cells not seen in training fall back to team-season REs
# via allow_new_levels = TRUE in the prediction calls.

prepare_scenario <- function(
    season,
    posteam,
    defteam,
    down,
    ydstogo,
    yardline_100,
    score_differential,
    qtr,
    game_seconds_remaining,
    half_seconds_remaining,
    offense_personnel_group,
    clock_stop               = 0,
    def_rate_Base            = 0.33,
    def_rate_Nickel          = 0.50,
    def_rate_Dime            = 0.17,
    off_rate_run             = 0.50,
    off_rate_pass            = 0.50,
    off_epa_by_personnel     = 0.00,
    off_success_by_personnel = 0.45,
    def_epa_allowed_Base     = 0.00,
    def_epa_allowed_Nickel   = 0.00,
    def_epa_allowed_Dime     = 0.00
) {
  packages <- c("Base", "Nickel", "Dime")

  # matchup_season mirrors the data_prep logic: only 2022+ get distinct levels
  season_int     <- as.integer(season)
  matchup_season <- ifelse(season_int >= 2022, as.character(season), "pre2022")

  base_row <- tibble(
    season                   = factor(season,  levels = levels(train_df$season)),
    posteam                  = factor(posteam, levels = levels(train_df$posteam)),
    defteam                  = factor(defteam, levels = levels(train_df$defteam)),
    off_team_season          = factor(paste0(posteam, "_", season),
                                      levels = levels(train_df$off_team_season)),
    def_team_season          = factor(paste0(defteam, "_", season),
                                      levels = levels(train_df$def_team_season)),
    down                     = down,
    ydstogo                  = ydstogo,
    yardline_100             = yardline_100,
    score_differential       = score_differential,
    qtr                      = factor(qtr, levels = levels(train_df$qtr)),
    game_seconds_remaining   = game_seconds_remaining,
    half_seconds_remaining   = half_seconds_remaining,
    clock_stop               = clock_stop,
    offense_personnel_group  = factor(offense_personnel_group,
                                      levels = levels(train_df$offense_personnel_group)),
    def_rate_Base            = def_rate_Base,
    def_rate_Nickel          = def_rate_Nickel,
    def_rate_Dime            = def_rate_Dime,
    off_rate_run             = off_rate_run,
    off_rate_pass            = off_rate_pass,
    off_epa_by_personnel     = off_epa_by_personnel,
    off_success_by_personnel = off_success_by_personnel,
    def_epa_allowed_Base     = def_epa_allowed_Base,
    def_epa_allowed_Nickel   = def_epa_allowed_Nickel,
    def_epa_allowed_Dime     = def_epa_allowed_Dime
  )

  scenario_df <- bind_rows(lapply(packages, function(pkg) {
    base_row %>%
      mutate(
        defense_personnel_group = factor(pkg,
          levels = c("Base", "Nickel", "Dime")),
        matchup_cell = factor(
          paste(paste0(defteam, "_", matchup_season),
                offense_personnel_group, pkg, sep = ":"),
          levels = levels(train_df$matchup_cell)
        ),
        off_personnel_team_season = factor(
          paste(paste0(posteam, "_", season),
                offense_personnel_group, sep = ":"),
          levels = levels(train_df$off_personnel_team_season)
        )
      )
  }))

  scenario_df <- scale_apply(scenario_df, train_means, train_sds, scale_vars)
  scenario_df
}


# ── recommend_personnel ───────────────────────────────────────────────────────
# Runs all three fitted models on the 3-package scenario, marginalises over
# the run/pass split per draw, and returns full posterior summaries.
#
# E[EPA | package] = P(pass|pkg) * E[EPA|pass,pkg]
#                  + P(run |pkg) * E[EPA|run, pkg]
#
# Computed across all posterior draws so uncertainty is fully propagated.

recommend_personnel <- function(fit_pass, fit_epa_pass, fit_epa_run,
                                 scenario_df) {
  packages <- as.character(scenario_df$defense_personnel_group)

  p_pass_draws <- posterior_epred(fit_pass,     newdata = scenario_df,
                                   allow_new_levels = TRUE)
  p_run_draws  <- 1 - p_pass_draws

  epa_pass_draws <- posterior_epred(fit_epa_pass, newdata = scenario_df,
                                     allow_new_levels = TRUE)
  epa_run_draws  <- posterior_epred(fit_epa_run,  newdata = scenario_df,
                                     allow_new_levels = TRUE)

  marginal_epa_draws <- p_pass_draws * epa_pass_draws +
                         p_run_draws  * epa_run_draws

  colnames(marginal_epa_draws) <- packages
  colnames(p_pass_draws)       <- packages
  colnames(epa_pass_draws)     <- packages
  colnames(epa_run_draws)      <- packages

  summary_tbl <- tibble(
    package        = packages,
    p_pass_mean    = colMeans(p_pass_draws),
    p_pass_lower80 = apply(p_pass_draws,       2, quantile, 0.10),
    p_pass_upper80 = apply(p_pass_draws,       2, quantile, 0.90),
    epa_pass_mean  = colMeans(epa_pass_draws),
    epa_run_mean   = colMeans(epa_run_draws),
    epa_mean       = colMeans(marginal_epa_draws),
    epa_sd         = apply(marginal_epa_draws, 2, sd),
    epa_lower80    = apply(marginal_epa_draws, 2, quantile, 0.10),
    epa_median     = apply(marginal_epa_draws, 2, quantile, 0.50),
    epa_upper80    = apply(marginal_epa_draws, 2, quantile, 0.90),
    epa_lower95    = apply(marginal_epa_draws, 2, quantile, 0.025),
    epa_upper95    = apply(marginal_epa_draws, 2, quantile, 0.975)
  )

  best_idx  <- apply(marginal_epa_draws, 1, which.min)
  prob_best <- tibble(
    package   = packages,
    prob_best = sapply(seq_along(packages),
                       function(i) mean(best_idx == i))
  )

  summary_tbl <- summary_tbl %>%
    left_join(prob_best, by = "package") %>%
    arrange(epa_mean)

  best_pkg   <- summary_tbl$package[1]
  second_pkg <- summary_tbl$package[2]
  edge_draws <- marginal_epa_draws[, second_pkg] - marginal_epa_draws[, best_pkg]

  list(
    summary          = summary_tbl,
    recommendation   = best_pkg,
    probability_best = summary_tbl$prob_best[1],
    expected_edge    = mean(edge_draws),
    edge_sd          = sd(edge_draws),
    edge_lower80     = quantile(edge_draws, 0.10),
    edge_upper80     = quantile(edge_draws, 0.90),
    edge_lower95     = quantile(edge_draws, 0.025),
    edge_upper95     = quantile(edge_draws, 0.975),
    draws            = marginal_epa_draws
  )
}


# ── print_recommendation ──────────────────────────────────────────────────────
# Full formatted output with confidence strength label.
#
# Confidence thresholds:
#   Strong Recommend — prob_best >= 0.75 AND 80% CI lower bound on edge > 0
#                      Both confident AND margin is reliably positive
#   Recommend        — prob_best >= 0.55 OR expected edge > 0.02
#                      Clear preference but meaningful uncertainty remains
#   Light Recommend  — everything else
#                      Distributions substantially overlap; use as one input

print_recommendation <- function(result) {

  best_row   <- result$summary[1, ]
  strength   <- classify_strength(
    result$probability_best,
    result$edge_lower80,
    result$expected_edge
  )

  strength_color <- switch(strength,
    "Strong Recommend" = "\033[1;32m",  # bold green
    "Recommend"        = "\033[1;33m",  # bold yellow
    "Light Recommend"  = "\033[1;36m"   # bold cyan
  )
  reset <- "\033[0m"

  cat("\n")
  cat("╔══════════════════════════════════════════════════════╗\n")
  cat("║         DEFENSIVE PERSONNEL RECOMMENDATION          ║\n")
  cat("╚══════════════════════════════════════════════════════╝\n\n")

  cat(sprintf("  PACKAGE:    %s\n", result$recommendation))
  cat(sprintf("  CONFIDENCE: %s%s%s\n\n",
              strength_color, strength, reset))

  cat("  ── EPA Distribution (recommended package) ──────────\n")
  cat(sprintf("  Expected EPA allowed:  %+.4f\n",  best_row$epa_mean))
  cat(sprintf("  Median EPA allowed:    %+.4f\n",  best_row$epa_median))
  cat(sprintf("  Std deviation:         %.4f\n",   best_row$epa_sd))
  cat(sprintf("  80%% credible interval: [%+.4f, %+.4f]\n",
              best_row$epa_lower80, best_row$epa_upper80))
  cat(sprintf("  95%% credible interval: [%+.4f, %+.4f]\n\n",
              best_row$epa_lower95, best_row$epa_upper95))

  cat("  ── Edge over second-best ────────────────────────────\n")
  cat(sprintf("  Expected edge:         %+.4f EPA\n",   result$expected_edge))
  cat(sprintf("  80%% CI on edge:        [%+.4f, %+.4f]\n",
              result$edge_lower80, result$edge_upper80))
  cat(sprintf("  Probability best:      %.1f%%\n\n",
              result$probability_best * 100))

  cat("  ── All packages ─────────────────────────────────────\n")
  tbl <- result$summary %>%
    mutate(
      `Package`   = package,
      `P(pass)`   = sprintf("%.1f%%", p_pass_mean * 100),
      `EPA|pass`  = sprintf("%+.3f",  epa_pass_mean),
      `EPA|run`   = sprintf("%+.3f",  epa_run_mean),
      `E[EPA]`    = sprintf("%+.3f",  epa_mean),
      `80% CI`    = sprintf("[%+.3f, %+.3f]", epa_lower80, epa_upper80),
      `P(best)`   = sprintf("%.1f%%", prob_best * 100)
    ) %>%
    select(`Package`, `P(pass)`, `EPA|pass`, `EPA|run`,
           `E[EPA]`, `80% CI`, `P(best)`)
  print(as.data.frame(tbl), row.names = FALSE)
  cat("\n")

  invisible(result)
}

Appendix: Confidence Threshold Logic

The three confidence levels are defined as follows and are designed to give a defensive coordinator actionable context alongside the recommendation:

Level Condition Interpretation
Strong Recommend prob_best >= 0.75 AND edge_lower80 > 0 Model is confident this is the best package AND confident the margin over the next-best option is meaningfully positive. Trust this call.
Recommend prob_best >= 0.55 OR expected_edge > 0.02 Clear preferred package but non-trivial uncertainty remains. Use as primary input alongside film and personnel considerations.
Light Recommend Everything else Posterior distributions substantially overlap. The model has a slight preference but the situation is genuinely close. Non-model factors — injuries, specific matchups, scheme — should weigh heavily.

The edge_lower80 condition in Strong Recommend is important: a model can be 75% confident Nickel is best while the 80% CI on the edge includes negative values, meaning there is meaningful posterior probability that Base is actually better by a small margin. Requiring both conditions ensures Strong Recommend is only applied when the model’s confidence is robust to posterior uncertainty, not just a reflection of a noisy plurality.

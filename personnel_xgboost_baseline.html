3. Derive Personnel Groupings

import re

def extract_offense_personnel(s):
    if not isinstance(s, str):
        return None
    rb = re.search(r'(\d+)\s*RB', s)
    te = re.search(r'(\d+)\s*TE', s)
    if rb and te:
        return str(int(rb.group(1))) + str(int(te.group(1)))
    return None

def extract_defense_personnel(s):
    if not isinstance(s, str):
        return None
    cb = re.search(r'(\d+)\s*CB', s)
    fs = re.search(r'(\d+)\s*FS', s)
    ss = re.search(r'(\d+)\s*SS', s)
    db_count = (
        int(cb.group(1)) if cb else 0
    ) + (
        int(fs.group(1)) if fs else 0
    ) + (
        int(ss.group(1)) if ss else 0
    )
    if db_count == 4:
        return "Base"
    elif db_count == 5:
        return "Nickel"
    elif db_count == 6:
        return "Dime"
    return None

raw["offense_personnel_group"] = raw["offense_personnel"].apply(extract_offense_personnel)
raw["defense_personnel_group"] = raw["defense_personnel"].apply(extract_defense_personnel)

9. Counterfactual Recommendation Function

def recommend_defense(
    down, ydstogo, yardline_100,
    score_differential, quarter_seconds_remaining,
    half_seconds_remaining, game_seconds_remaining,
    posteam, defteam, off_personnel
):
    packages = ["Base", "Nickel", "Dime"]
    results = {}

    for pkg in packages:
        try:
            posteam_enc = le_team.transform([posteam])[0]
            defteam_enc = le_team.transform([defteam])[0]
        except ValueError:
            print("Unknown team — check team abbreviation.")
            return

        off_enc = le_off.transform([off_personnel])[0] \
            if off_personnel in le_off.classes_ else -1
        def_enc = le_def.transform([pkg])[0]

        row = pd.DataFrame([{
            "down": down,
            "ydstogo": ydstogo,
            "yardline_100": yardline_100,
            "score_differential": score_differential,
            "quarter_seconds_remaining": quarter_seconds_remaining,
            "half_seconds_remaining": half_seconds_remaining,
            "game_seconds_remaining": game_seconds_remaining,
            "posteam_enc": posteam_enc,
            "defteam_enc": defteam_enc,
            "off_pers_enc": off_enc,
            "def_pers_enc": def_enc
        }])

        p_pass    = model_platype.predict_proba(row)[0][1]
        epa_run   = model_run.predict(row)[0]
        epa_pass  = model_pass.predict(row)[0]
        epa_exp   = p_pass * epa_pass + (1 - p_pass) * epa_run

        results[pkg] = {
            "Expected EPA": round(epa_exp, 4),
            "Run EPA":       round(epa_run, 4),
            "Pass EPA":      round(epa_pass, 4),
            "P(Pass)":       round(p_pass, 4)
        }

    results_df = pd.DataFrame(results).T.sort_values("Expected EPA")
    best = results_df.index[0]
    second = results_df.index[1]
    margin = results_df.loc[second, "Expected EPA"] - results_df.loc[best, "Expected EPA"]

    print(f"\n{'='*50}")
    print(f"  RECOMMENDATION: {best}")
    print(f"  Margin over next best ({second}): {margin:.4f} EPA")
    print(f"{'='*50}\n")
    print(results_df.to_string())
    return results_df
